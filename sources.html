API REST con Flask

Python

Crear una API REST simple utilizando Flask para gestionar una lista de tareas. 

from flask import Flask, jsonify, request

app = Flask(__name__)

tasks = []

@app.route('/tasks', methods=['GET'])
def get_tasks():
    return jsonify(tasks)

@app.route('/tasks', methods=['POST'])
def add_task():
    task = request.json
    tasks.append(task)
    return jsonify(task), 201

@app.route('/tasks/<int:task_id>', methods=['DELETE'])
def delete_task(task_id):
    if task_id < len(tasks):
        deleted_task = tasks.pop(task_id)
        return jsonify(deleted_task)
    return jsonify({'error': 'Task not found'}), 404

if __name__ == '__main__':
    app.run(debug=True)

Web Scraping con Python

Web

Extraer datos de páginas web utilizando BeautifulSoup y requests. 

import requests
from bs4 import BeautifulSoup
import csv

# Hacer la solicitud HTTP
url = 'https://ejemplo.com'
response = requests.get(url)

# Verificar que la solicitud fue exitosa
if response.status_code == 200:
    # Parsear el contenido HTML
    soup = BeautifulSoup(response.content, 'html.parser')
    
    # Encontrar elementos por etiqueta y clase
    titulos = soup.find_all('h2', class_='titulo')
    
    # Extraer texto de los elementos
    datos = []
    for titulo in titulos:
        datos.append(titulo.get_text().strip())
    
    # Guardar en CSV
    with open('datos_extraidos.csv', 'w', newline='', encoding='utf-8') as file:
        writer = csv.writer(file)
        writer.writerow(['Títulos'])
        for dato in datos:
            writer.writerow([dato])
    
    print("Datos extraídos y guardados correctamente")
else:
    print(f"Error al acceder a la página: {response.status_code}")
